```swift
import SwiftUI
import SwiftData

struct QuickQuoteView: View {
    @Environment(\.modelContext) private var modelContext
    @State private var jobTitle = ""
    @State private var clientName = ""
    @State private var lineItems: [LineItemDraft] = [LineItemDraft()]
    @State private var taxRate: Decimal = 0.0
    @FocusState private var focusedField: Field?
    
    enum Field: Hashable {
        case jobTitle, clientName, itemDesc, itemQty, itemPrice
    }
    
    var subtotal: Decimal {
        lineItems.reduce(0) { $0 + $1.quantity * $1.unitPrice }
    }
    
    var taxAmount: Decimal {
        lineItems.filter(\.taxable).reduce(0) { $0 + $1.quantity * $1.unitPrice } * (taxRate / 100)
    }
    
    var total: Decimal { subtotal + taxAmount }
    
    var body: some View {
        NavigationStack {
            Form {
                Section("Job Details") {
                    TextField("What's the job?", text: $jobTitle)
                        .focused($focusedField, equals: .jobTitle)
                        .font(.headline)
                        .submitLabel(.next)
                    
                    TextField("Client name", text: $clientName)
                        .focused($focusedField, equals: .clientName)
                        .submitLabel(.next)
                }
                
                Section("Line Items") {
                    ForEach($lineItems) { $item in
                        LineItemRow(item: $item, focusedField: $focusedField)
                    }
                    .onDelete { indexSet in
                        lineItems.remove(atOffsets: indexSet)
                    }
                    
                    Button {
                        withAnimation { lineItems.append(LineItemDraft()) }
                    } label: {
                        Label("Add Item", systemImage: "plus.circle.fill")
                    }
                }
                
                Section("Total") {
                    LabeledContent("Subtotal", value: subtotal, format: .currency(code: "USD"))
                    LabeledContent("Tax (\(taxRate)%)", value: taxAmount, format: .currency(code: "USD"))
                    LabeledContent("Total", value: total, format: .currency(code: .USD"))
                        .font(.headline)
                }
            }
            .navigationTitle("Quick Quote")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .confirmationAction) {
                    Button("Send") { saveAndSendQuote() }
                        .disabled(jobTitle.isEmpty || clientName.isEmpty)
                        .fontWeight(.semibold)
                }
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
            }
            .onAppear { focusedField = .jobTitle }
        }
    }
    
    private func saveAndSendQuote() {
        let client = Client(name: clientName)
        modelContext.insert(client)
        
        let job = Job(title: jobTitle, jobType: .quote, client: client)
        job.totalAmount = total
        
        for draft in lineItems {
            let item = LineItem(
                description: draft.description,
                quantity: draft.quantity,
                unitPrice: draft.unitPrice,
                taxable: draft.taxable
            )
            item.job = job
            modelContext.insert(item)
        }
        
        modelContext.insert(job)
        
        let generator = PDFInvoiceGenerator()
        let pdfData = generator.generate(job: job, client: client, taxRate: taxRate)
        PDFShareHelper.share(pdfData: pdfData, jobTitle: jobTitle)
    }
}

struct LineItemDraft: Identifiable {
    let id = UUID()
    var description: String = ""
    var quantity: Decimal = 1
    var unitPrice: Decimal = 0
    var taxable: Bool = true
}
```

#### 3.3.2 PDF发票生成器

```swift
import PDFKit

final class PDFInvoiceGenerator {
    func generate(job: Job, client: Client, taxRate: Decimal) -> Data {
        let pageWidth: CGFloat = 612
        let pageHeight: CGFloat = 792
        let margin: CGFloat = 50
        
        let pageRect = CGRect(x: 0, y: 0, width: pageWidth, height: pageHeight)
        let renderer = UIGraphicsPDFRenderer(bounds: pageRect)
        
        let data = renderer.pdfData { context in
            context.beginPage()
            
            var y: CGFloat = margin
            
            let headerAttrs: [NSAttributedString.Key: Any] = [
                .font: UIFont.systemFont(ofSize: 28, weight: .bold),
                .foregroundColor: UIColor.systemBlue
            ]
            let titleAttrs: [NSAttributedString.Key: Any] = [
                .font: UIFont.systemFont(ofSize: 14, weight: .medium),
                .foregroundColor: UIColor.darkGray
            ]
            let bodyAttrs: [NSAttributedString.Key: Any] = [
                .font: UIFont.systemFont(ofSize: 12),
                .foregroundColor: UIColor.black
            ]
            
            ("INVOICE" as NSString).draw(at: CGPoint(x: margin, y: y), withAttributes: headerAttrs)
            y += 40
            
            ("Job: \(job.title)" as NSString).draw(at: CGPoint(x: margin, y: y), withAttributes: titleAttrs)
            y += 20
            ("Date: \(job.createdAt.formatted(date: .abbreviated, time: .omitted))" as NSString)
                .draw(at: CGPoint(x: margin, y: y), withAttributes: bodyAttrs)
            y += 20
            ("Bill To: \(client.name)" as NSString).draw(at: CGPoint(x: margin, y: y), withAttributes: bodyAttrs)
            if let email = client.email {
                y += 18
                ("Email: \(email)" as NSString).draw(at: CGPoint(x: margin, y: y), withAttributes: bodyAttrs)
            }
            y += 30
            
            let tableX: CGFloat = margin
            let colWidths: [CGFloat] = [250, 80, 100, 100]
            let rowHeight: CGFloat = 25
            let headers = ["Description", "Qty", "Unit Price", "Total"]
            
            var x = tableX
            for (i, header) in headers.enumerated() {
                let rect = CGRect(x: x, y: y, width: colWidths[i], height: rowHeight)
                let headerBg = UIBezierPath(rect: rect)
                UIColor.systemBlue.setFill()
                headerBg.fill()
                (header as NSString).draw(in: rect.insetBy(dx: 5, dy: 4), withAttributes: [
                    .font: UIFont.systemFont(ofSize: 11, weight: .bold),
                    .foregroundColor: UIColor.white
                ])
                x += colWidths[i]
            }
            y += rowHeight
            
            for item in job.lineItems.sorted(by: { $0.sortOrder < $1.sortOrder }) {
                x = tableX
                let values = [
                    item.description,
                    "\(item.quantity)",
                    item.unitPrice.formatted(.currency(code: "USD")),
                    item.total.formatted(.currency(code: "USD"))
                ]
                for (i, value) in values.enumerated() {
                    let rect = CGRect(x: x, y: y, width: colWidths[i], height: rowHeight)
                    (value as NSString).draw(in: rect.insetBy(dx: 5, dy: 4), withAttributes: bodyAttrs)
                    x += colWidths[i]
                }
                y += rowHeight
            }
            
            y += 10
            let subtotal = job.lineItems.reduce(Decimal(0)) { $0 + $1.total }
            let taxableTotal = job.lineItems.filter(\.taxable).reduce(Decimal(0)) { $0 + $1.total }
            let tax = taxableTotal * (taxRate / 100)
            let total = subtotal + tax
            
            let totals: [(String, String)] = [
                ("Subtotal", subtotal.formatted(.currency(code: "USD"))),
                ("Tax (\(taxRate)%)", tax.formatted(.currency(code: "USD"))),
                ("TOTAL", total.formatted(.currency(code: "USD")))
            ]
            
            for (label, value) in totals {
                let labelRect = CGRect(x: pageWidth - margin - 250, y: y, width: 150, height: rowHeight)
                let valueRect = CGRect(x: pageWidth - margin - 100, y: y, width: 100, height: rowHeight)
                let attrs = label == "TOTAL" ? [
                    .font: UIFont.systemFont(ofSize: 14, weight: .bold),
                    .foregroundColor: UIColor.black
                ] : bodyAttrs
                (label as NSString).draw(in: labelRect.insetBy(dx: 5, dy: 4), withAttributes: attrs)
                (value as NSString).draw(in: valueRect.insetBy(dx: 5, dy: 4), withAttributes: attrs)
                y += rowHeight
            }
            
            y += 30
            let footerAttrs: [NSAttributedString.Key: Any] = [
                .font: UIFont.systemFont(ofSize: 10),
                .foregroundColor: UIColor.gray
            ]
            ("Generated by JobRig — Quote · Invoice · Get Paid" as NSString)
                .draw(at: CGPoint(x: margin, y: pageHeight - margin), withAttributes: footerAttrs)
        }
        
        return data
    }
}
```

```swift
extension Job {
    func convertToInvoice(in context: ModelContext) -> Job {
        let invoice = Job(title: self.title, jobType: .invoice, client: self.client)
        invoice.totalAmount = self.totalAmount
        invoice.status = .invoiceSent
        
        for item in self.lineItems {
            let invoiceItem = LineItem(
                description: item.description,
                quantity: item.quantity,
                unitPrice: item.unitPrice,
                taxable: item.taxable
            )
            invoiceItem.sortOrder = item.sortOrder
            invoiceItem.job = invoice
            context.insert(invoiceItem)
        }
        
        for photo in self.photos {
            let invoicePhoto = JobPhoto(imageData: photo.imageData, caption: photo.caption)
            invoicePhoto.job = invoice
            context.insert(invoicePhoto)
        }
        
        context.insert(invoice)
        return invoice
    }
}
```

#### 3.3.4 离线优先数据同步

```swift
import CloudKit
import SwiftData

final class SyncManager: ObservableObject {
    static let shared = SyncManager()
    
    private let container = CKContainer(identifier: "iCloud.com.jobrig.app")
    private let privateDB: CKDatabase
    
    @Published var syncStatus: SyncStatus = .idle
    @Published var lastSyncDate: Date?
    
    enum SyncStatus {
        case idle
        case syncing
        case failed(Error)
        case completed
    }
    
    init() {
        self.privateDB = container.privateCloudDatabase
    }
    
    func pushLocalChanges(modelContext: ModelContext) async {
        guard syncStatus != .syncing else { return }
        await MainActor.run { syncStatus = .syncing }
        
        do {
            let jobs = try modelContext.fetch(FetchDescriptor<Job>(
                predicate: #Predicate { $0.updatedAt > lastSyncDate ?? .distantPast }
            ))
            
            for job in jobs {
                let record = try job.toCKRecord()
                try await privateDB.save(record)
            }
            
            await MainActor.run {
                lastSyncDate = Date()
                syncStatus = .completed
            }
        } catch {
            await MainActor.run { syncStatus = .failed(error) }
        }
    }
    
    func pullRemoteChanges(modelContext: ModelContext) async {
        guard syncStatus != .syncing else { return }
        await MainActor.run { syncStatus = .syncing }
        
        do {
            let zone = CKRecordZone.default()
            var changedRecords: [CKRecord] = []
            var cursor: CKQueryOperation.Cursor?
            
            let query = CKQuery(recordType: "CD_Job", predicate: NSPredicate(value: true))
            let (results, nextCursor) = try await privateDB.records(matching: query)
            
            for (_, result) in results {
                if case .success(let record) = result {
                    changedRecords.append(record)
                }
            }
            cursor = nextCursor
            
            for record in changedRecords {
                try mergeRecord(record, into: modelContext)
            }
            
            try modelContext.save()
            await MainActor.run {
                lastSyncDate = Date()
                syncStatus = .completed
            }
        } catch {
            await MainActor.run { syncStatus = .failed(error) }
        }
    }
    
    private func mergeRecord(_ record: CKRecord, into context: ModelContext) throws {
        let id = record["id"] as? String ?? UUID().uuidString
        let descriptor = FetchDescriptor<Job>(predicate: #Predicate { $0.id.uuidString == id })
        let existing = try context.fetch(descriptor).first
        
        if let existing = existing {
            existing.title = record["title"] as? String ?? existing.title
            existing.status = JobStatus(rawValue: record["status"] as? String ?? "") ?? existing.status
            existing.totalAmount = Decimal(record["totalAmount"] as? Double ?? Double(exexisting.totalAmount))
            existing.updatedAt = record.modificationDate ?? Date()
        }
    }
}
```

```swift
import StoreKit

@Observable
final class SubscriptionManager {
    var isSubscribed: Bool = false
    var subscriptionTier: Tier = .free
    var products: [Product] = []
    
    enum Tier: String, CaseIterable {
        case free = "free"
        case pro = "com.jobrig.pro.monthly"
        case proAnnual = "com.jobrig.pro.annual"
    }
    
    var monthlyPrice: String? {
        products.first(where: { $0.id == Tier.pro.rawValue })?.displayPrice
    }
    
    var annualPrice: String? {
        products.first(where: { $0.id == Tier.proAnnual.rawValue })?.displayPrice
    }
    
    func loadProducts() async {
        do {
            products = try await Product.products(for: [
                Tier.pro.rawValue,
                Tier.proAnnual.rawValue
            ])
        } catch {
            print("Failed to load products: \(error)")
        }
    }
    
    func checkSubscriptionStatus() async {
        for await result in Transaction.currentEntitlements {
            if case .verified(let transaction) = result {
                if transaction.productID == Tier.pro.rawValue ||
                    transaction.productID == Tier.proAnnual.rawValue {
                    isSubscribed = transaction.expirationDate?.compare(Date()) == .orderedDescending
                    subscriptionTier = .pro
                    return
                }
            }
        }
        isSubscribed = false
        subscriptionTier = .free
    }
    
    func purchase(_ product: Product) async throws -> StoreKit.Transaction? {
        let result = try await product.purchase()
        
        switch result {
        case .success(let verification):
            let transaction = try checkVerified(verification)
            await transaction.finish()
            await checkSubscriptionStatus()
            return transaction
        case .pending, .userCancelled:
            return nil
        @unknown default:
            return nil
        }
    }
    
    private func checkVerified(_ result: VerificationResult<StoreKit.Transaction>) throws -> StoreKit.Transaction {
        switch result {
        case .unverified(_, let error):
            throw error
        case .verified(let transaction):
            return transaction
        }
    }
}
```

### 7.1 定价策略深度分析

1. 承包商工作追踪器功能相对聚焦（报价+发票+客户追踪），不需要复杂的AI生成费用

3. 免费下载+免费基础功能可以最大化用户获取（ASO排名靠下载量驱动）
4. 一次性订阅买断对用户最有吸引力（承包商讨厌月费）
